﻿/**
 * This used to manage the Logout related actvities
 * @date : 11-08-2016
 * @lastModified: 11-08-2016
 * @Author : Goutham KD
 * @param $q
 * @param $cookieStore
 * @param $state
 */
 
(function (angular) {
 
    "use strict";
 
    angular.module('dataVINEApp').service('logoutservice', ['$q', '$sessionStorage', '$state', 'NetworkUtilService',
        function ($q, $sessionStorage, $state, networkUtilService) {
            this.logout = function () {
                var des = $q.defer();
                var arrInputData = [];
                networkUtilService.createHttpRequestAndGetResponse("logout", arrInputData).then(function (objData) {
                    $sessionStorage.remove("loginName");
                    $sessionStorage.remove("rememberme");
                    $sessionStorage.remove("superUser");
                    des.resolve(objData);
                });
                return des.promise;
            };
 
        }]);
})(window.angular);
