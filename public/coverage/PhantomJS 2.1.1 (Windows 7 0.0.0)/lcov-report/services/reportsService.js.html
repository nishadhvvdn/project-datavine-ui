 * @param $q
 * @param networkUtilService

(function (angular) {
 
    "use strict";
 
    angular.module('dataVINEApp').service('reportsService', ['$q', 'NetworkUtilService',
        function ($q, networkUtilService) {
            this.communicationStatistics = function (endpoint,isHyperhub) {
                var des = $q.defer();
                var arrInputData = (isHyperhub===undefined)?[]:[isHyperhub];
                networkUtilService.createHttpRequestAndGetResponse(endpoint,arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
            this.NewAccountReport = function (fromDate,toDate) {
                var des = $q.defer();
                var arrInputData=[fromDate,toDate];
                networkUtilService.createHttpRequestAndGetResponse("NewAccountReport",arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
            this.datavineHealth = function (param) {
                var des = $q.defer();
                var arrInputData = [param];
                networkUtilService.createHttpRequestAndGetResponse("DataVINEHealthReport", arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
            this.systemLog = function (firstDate, lastDate) {
                var des = $q.defer();
                var arrInputData = [firstDate, lastDate];
                networkUtilService.createHttpRequestAndGetResponse("SystemLogReport", arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
            this.deviceFirmwareVersions = function () {
                var des = $q.defer();
                networkUtilService.createHttpRequestAndGetResponse("DeviceFirmwareVersionReport").then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
            this.deviceRegistrationStatus = function () {
                var des = $q.defer();
                networkUtilService.createHttpRequestAndGetResponse("deviceRegistrationStatus").then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
            this.auditLog = function (firstDate, lastDate) {
                var des = $q.defer();
                var arrInputData = [firstDate, lastDate];
                networkUtilService.createHttpRequestAndGetResponse("SystemAuditLogReport", arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
        }]);
})(window.angular);
