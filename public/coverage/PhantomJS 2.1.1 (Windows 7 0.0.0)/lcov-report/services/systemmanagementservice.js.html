/**
 * This handles System management services
 * @date : 11-08-2016
 * @lastModified: 11-08-2016
 * @Author : Ganesh
 * @param $q
 * @param networkUtilService
 */
 
(function (angular) {
 
    "use strict";
 
    angular.module('dataVINEApp').service('SystemManagementService', ['$q', 'NetworkUtilService',
        function ($q, networkUtilService) {
 
            this.getJobStatus = function () {
                var des = $q.defer();
                networkUtilService.createHttpRequestAndGetResponse("SMNetworkJobStatus").then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
 
            this.getSMHypersproutDetails = function (param,type) {
                var des = $q.defer();
                var arrInputData = [param,type];
                networkUtilService.createHttpRequestAndGetResponse("SMHyperSprout", arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
 
            this.getSMMeterDetails = function (param) {
                var des = $q.defer();
                var arrInputData = [param];
                networkUtilService.createHttpRequestAndGetResponse("SMMeters", arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
 
            this.getNodePingDetails = function (serialId, type) {
                var des = $q.defer();
                var arrInputData = [serialId, type];
                networkUtilService.createHttpRequestAndGetResponse((type === 'HyperSprout') ? 'SMNodePing' : 'SMMeterNodePing', arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
        }]);
})(window.angular);
