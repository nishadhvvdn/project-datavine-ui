'use strict';
 
/**
 * This handles administration services
 * @date : 29-09-2016
 * @lastModified: 29-09-2016
 * @Author : Sunil
 * @param $q
 * @param networkUtilService
 */
 
(function (angular) {
 
    "use strict";
 
    angular.module('dataVINEApp').service('toolsService', ['$q', 'NetworkUtilService',
        function ($q, networkUtilService) {
            this.UpdateUserSettings = function (userID, firstName, lastName, email, homePage, timeZone, oldPswd, loginId, temprature) {
                var des = $q.defer();
                var arrInputData = [userID, firstName, lastName, email, homePage, timeZone, oldPswd, loginId, temprature];
                networkUtilService.createHttpRequestAndGetResponse("UpdateUserSettings", arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
 
            this.UserSettings = function (userID) {
                var des = $q.defer();
                var arrInputData = [userID];
                networkUtilService.createHttpRequestAndGetResponse("UserSettings", arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
            this.ChangePassword = function (userID, OldPassword, NewPassword, loginId) {
                var des = $q.defer();
                var arrInputData = [userID, OldPassword, NewPassword, loginId];
                networkUtilService.createHttpRequestAndGetResponse("ChangePassword", arrInputData).then(function (objData) {
                    des.resolve(objData);
                });
                return des.promise;
            };
        }]);
})(window.angular);
